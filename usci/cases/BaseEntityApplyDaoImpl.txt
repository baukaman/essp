applyComplexSet, case#7
---------------------------------------------------------

#               TEAM
#               /   \
#              /   CODE
#             /
#    users  [ (user),(user),(user)]
#                |
#        quote, relations
#        /        |
#     string   (user_id, relation_type_id)
#
#

#====================================
#= RELATION
#====================================
meta create --name=relation
meta add --name=relation --attribute=relation_type_id --type=INTEGER --final
meta add --name=relation --attribute=user_id --type=INTEGER --final


#====================================
#= USER
#====================================
meta create --name=user
meta add --name=user --attribute=relations --type=meta_class --array --childname=relation --final
meta add --name=user --attribute=quote --type=string --final

#====================================
#= TEAM
#====================================
meta create --name=team
meta add --name=team --attribute=users --type=meta_class --array --childname=user --final
meta add --name=team --attribute=code --type=string
meta key --name=team --attribute=code
